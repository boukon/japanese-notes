# ~/.bashrc
#!/bin/bash
PSmoji=($(
# すごい漢字

#           ﾋﾂﾖｳ
# 例語（１）必要「necessary」や「essential」
#           ｶﾅﾗ
# 例語（２）必ず「inevitable」や「certain」
printf 必' '

#           ｷｮｳｾｲ
# 例語（１）凶星「ominous star」や「star of bad omen」や「malevolent stars」
#           ｷｮｳｱｸ
# 例語（２）凶悪「heinous」や「terrible(ie. causing terror)」
printf 凶' '

#                   ｸｽﾘ
# 漢字の読み方      薬
#
#           ﾔｸﾌﾞﾂ
# 例語（１）薬物「medicine」や「drugs」
#
# 例文、
# 「自分の異形________は俺の____________っす。」
# 前行の英語意訳は、
# 「My personal _______ is my ____________.」
# 英語メモ、the above sentence is informal and would have
#  a male speaker. I omitted the term 異形 from the translation;
#  it implies that the ____ is a grotesque variant
#  of the object in question: ________, ________. In English, I 
#  believe this is already conveyed through "[m]y personal" which
#  came from 自分の, literally "myself's," and the dark subject
#  matter.
#  This is due to the use of 俺 and っす as opposed to,
#  say, 僕 or 私 and です。
printf 薬' '

# 苦しい「suffering」          (kanji) (radical form)
# 英語メモ、old (古) + vegetation (艸 -> 艹)
#  In Buddhism, 苦 refers to suffering: the concept 
#  known in the West more often as Dukkha.
printf 苦' '

# 例、バンド「dormouseの森」、英語は「forest of dormouse」
printf 森' '

# 悪い「wicked」
# 悪夢「nightmare」
# 悪魔「devil」
printf 悪' '

# 自決「self-determination」と「suicide」（自殺に類義語）
# 決意「self-determination」や「personal resolution」（自殺ない）
printf 決' '

# 核殺「Coakira!! :D」
# 核（一語）の英語は「nucleus」です。
# この語のその他の用法は「core」です。
printf 核' '

# 異形「unnerving」や「unnatural」でも「grotesque」
printf 異' '

# 火「fire」
# 炎「flame, blaze, fire; intense emotion, passion」
# 焱「flames, blaze; cracking of flames」
# 燚「flames; blaze; scorch」
printf 炎' '

# 全然の英語は「(not) at all」と「totally, completely」です。
# 必然的の英語は「inevitability」や「certainty」です。
#  僕の未来に___を見よ。（それは暗いすぎるね？ｗｗ）
# 必然性は前の語の同義語です。
# 未然の用法は「before something happens」です。
#  未然にの英語は「before something happens」や「beforehand」です。
#  未然に防ぐの英語は「prevent」や「nip in the bud」です。
printf 然' '

# 必殺の英語は「deadliness, lethality」です。
printf 殺' '

# harsh noise band 非常階段！！emergency staircase!
# 非常の英語直訳は「not usual」でも翻訳は「emergency」です。
printf 非' '

# 断罪の英語は「great sin」と「decapitation as punishment for a crime」です。
printf 罪' '

# ___の英語は「____________」です。
# ______為の英語は「______」です。
printf 自' '
# ___の英語は「____________」です。
# ______為の英語は「______」です。
printf __' '
# ___の英語は「____________」です。
# ______為の英語は「______」です。。
printf 癖' '

# 暁の英語は「daybreak」です。
# 例は同人サークル「暁Records」です。
printf 暁' '
))
PSchar() {
 [ "$PWD" == "$HOME" ] && 
  printf %s 〜 ||
 printf %s ${PSmoji[$((RANDOM%${#PSmoji[@]}))]}
}
PS1='\[\e[2m\]$(PSchar)\[\e[0m\]'
